require "ifj21"

function fibo(param1: integer, param2: integer): number
    return param1/1 + param2/1
end

function main(param1: integer)
    local a : integer = 1
    local b : integer = 1
    while param1 ~= 0 do
        local c : number = fibo(a, b)
        a = b
        b = tointeger(c)
        write (b)
        param1 = param1 - 1
    end
end

main(10)
