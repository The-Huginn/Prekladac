require "ifj21"

function main()
    local a : integer = readi()
    local sum : integer = 0
    local current : integer = 1
    repeat
        sum = sum * current
        current = current + 1
        if (sum >= 100) then
            break
        end
    until a 
    write("Faktorial ",a," je s breakom: ",sum,"\n")
end

main()
