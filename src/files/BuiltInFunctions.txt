LABEL substr1
	
	CREATEFRAME
	PUSHFRAME
	DEFVAR LF@string
	POPS LF@string
	DEFVAR LF@i
	POPS LF@i
	DEFVAR LF@j
	POPS LF@j
	DEFVAR LF@temp
	DEFVAR LF@result
	TYPE LF@temp LF@string
	JUMPIFEQ substrerror1 string@nil LF@temp
	TYPE LF@temp LF@i
	JUMPIFEQ substrerror1 string@nil LF@temp
	TYPE LF@temp LF@j
	JUMPIFEQ substrerror1 string@nil LF@temp
	DEFVAR LF@check
	DEFVAR LF@len
	STRLEN LF@len LF@string
	LT LF@check LF@i int@1
	JUMPIFEQ substrempty1 LF@check bool@true
	LT LF@check LF@j int@1
	JUMPIFEQ substrempty1 LF@check bool@true
	GT LF@check LF@i LF@len
	JUMPIFEQ substrempty1 LF@check bool@true
	GT LF@check LF@j LF@len
	JUMPIFEQ substrempty1 LF@check bool@true
	GT LF@check LF@i LF@j
	JUMPIFEQ substrempty1 LF@check bool@true
	MOVE LF@len LF@i
	SUB LF@len LF@len int@1
	MOVE LF@result string@
LABEL substrloop1
	GETCHAR LF@temp LF@string LF@len
	CONCAT LF@result LF@result LF@temp
	ADD LF@len LF@len int@1
	JUMPIFEQ substrpush1 LF@len LF@j
	JUMP substrloop1

LABEL substrerror1
	EXIT int@8
	
LABEL substrempty1
	MOVE LF@result string@
	
LABEL substrpush1
	PUSHS LF@result
	POPFRAME
	RETURN

LABEL write1

PUSHFRAME
DEFVAR LF@n
POPS LF@n
DEFVAR LF@term
LABEL write_loop1

SUB LF@n LF@n int@1
JUMPIFNEQ write_loop1 LF@n int@0
POPFRAME
RETURN

LABEL chr1
CREATEFRAME
PUSHFRAME
DEFVAR LF@i
POPS LF@i
DEFVAR LF@check
TYPE LF@check LF@i
JUMPIFEQ chrerror1 LF@check string@nil
DEFVAR LF@temp
LT LF@temp LF@i int@0
JUMPIFEQ chrnil1 LF@temp bool@true
GT LF@temp LF@i int@255
JUMPIFEQ chrnil1 LF@temp bool@true
DEFVAR LF@vysl
INT2CHAR LF@vysl LF@i
PUSHS LF@vysl
POPFRAME
RETURN
LABEL chrnil1
PUSHS nil@nil
POPFRAME
RETURN
LABEL chrerror1
EXIT int@8

LABEL ord1
CREATEFRAME
PUSHFRAME
DEFVAR LF@str
POPS LF@str
DEFVAR LF@i
POPS LF@i
DEFVAR LF@check
TYPE LF@check LF@i
JUMPIFEQ orderror1 LF@check string@nil
TYPE LF@check LF@str
JUMPIFEQ orderror1 LF@check string@nil
DEFVAR LF@len
STRLEN LF@len LF@str
DEFVAR LF@temp
LT LF@temp LF@i int@1
JUMPIFEQ ordnil1 LF@temp bool@true
GT LF@temp LF@i LF@len
JUMPIFEQ ordnil1 LF@temp bool@true
SUB LF@i LF@i int@1
DEFVAR LF@result
STRI2INT LF@result LF@str LF@i
PUSHS LF@result
POPFRAME
RETURN
LABEL ordnil1
PUSHS nil@nil
POPFRAME
RETURN
LABEL orderror1
